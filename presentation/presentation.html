Algoritmo de Scoring


def calculate_zone_score(zone_data, user_preferences):
    weights = {
        'security': user_preferences.get('security_weight', 0.25),
        'transport': user_preferences.get('transport_weight', 0.20),
        'cost': user_preferences.get('cost_weight', 0.20),
        'amenities': user_preferences.get('amenities_weight', 0.15),
        'education': user_preferences.get('education_weight', 0.10),
        'healthcare': user_preferences.get('healthcare_weight', 0.10)
    }
    
    score = sum(zone_data[factor] * weight 
                for factor, weight in weights.items())
    
    return min(max(score, 0), 10)  # Normalize to 0-10
